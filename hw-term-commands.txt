CREATE DATABASE Bamazon;
USE Bamazon;
CREATE TABLE products (
itemID int AUTO_INCREMENT,
productName varchar(30) NOT NULL,
departmentName varchar(30) NOT NULL,
price varchar(30) NOT NULL,
stockQuantity int NOT NULL,
PRIMARY KEY(itemID)
);
